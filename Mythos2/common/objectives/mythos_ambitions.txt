# Dragon Purity Ambitions
# Not normally selectable, but given by event

### Purity from Gold ###

# Build a horde of 200 gold
obj_build_horde_1 = {
	type = character
	
	ai_greed = 30
	
	potential = {
		has_character_modifier = can_ambition_dragon_gold_1
		OR = {
			religion = draconic_pagan
			religion = draconic_pagan_reformed
		}
	}
	
	allow = {
		has_character_modifier = can_ambition_dragon_gold_1
		OR = {
			religion = draconic_pagan
			religion = draconic_pagan_reformed
		}
	}
	
	chance = {
		factor = 100
		
		modifier = {
			factor = 1.5
			trait = zealous
		}
		modifier = {
			factor = 10
			trait = greedy
		}
		modifier = {
			factor = 1.25
			trait = patient
		}
		modifier = {
			factor = 1.25
			trait = fortune_builder
		}
		modifier = {
			factor = 1.25
			trait = midas_touched
		}
	}
	success = {
		wealth = 200
	}
	abort = {
		NOR = {
			is_incapable = no
			religion = draconic_pagan
			religion = draconic_pagan_reformed
		}
	}
	abort_effect = {
	}
	effect = {
		add_character_modifier = { modifier = built_horde years = 1 }
		hidden_tooltip = {
			set_character_flag = finished_dragon_ambition_gold_1
			remove_character_modifier = purity_review_timer
			remove_character_modifier = can_ambition_dragon_gold_1
			character_event = { id = DRAGON.56 }
		}
	}
}

### Purity from Piety ###

# Gain 2000 piety
obj_become_pious_1 = {
	type = character
	
	ai_zeal = 30
	
	potential = {
		has_character_modifier = can_ambition_dragon_piety_1
		OR = {
			religion = draconic_pagan
			religion = draconic_pagan_reformed
		}
	}
	
	allow = {
		has_character_modifier = can_ambition_dragon_piety_1
		OR = {
			religion = draconic_pagan
			religion = draconic_pagan_reformed
		}
	}
	
	chance = {
		factor = 100
		
		modifier = {
			factor = 10
			trait = zealous
		}
		modifier = {
			factor = 2
			trait = diligent
		}
		modifier = {
			factor = 0.01
			trait = cynical
		}
		modifier = {
			factor = 10.0
			OR = {
				trait = detached_priest
				trait = martial_cleric
				trait = scholarly_theologian
				trait = mastermind_theologian
			}
		}
		modifier = {
			factor = 0.01
			# Can use paragon of virtue instead
			NOT = { piety = 100 }
			is_ruler = yes
			NOT = { trait = cruel }
			NOT = { trait = impaler }
			NOT = { trait = incapable }
		}
	}
	success = {
		piety = 2000
	}
	abort = {
		NOR = {
			is_incapable = no
			religion = draconic_pagan
			religion = draconic_pagan_reformed
		}
	}
	abort_effect = {
	}
	effect = {
		add_character_modifier = { modifier = the_pious years = 1 }
		hidden_tooltip = {
			set_character_flag = finished_dragon_ambition_piety_1
			remove_character_modifier = purity_review_timer
			remove_character_modifier = can_ambition_dragon_piety_1
			character_event = { id = DRAGON.56 }
		}
	}
}
