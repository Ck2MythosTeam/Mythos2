# You are being challenged due to not being a known werewolf or not being werewolf culture
trigger_minor_challenge = {
	age = 16
	# Not challenged over being female
	OR = {
		is_female = no
		has_law = werewolf_law_4
		AND = {
			has_dlc = "Zeus"
			NOT = { has_law = status_of_women_0 }
		}
		NAND = {
			has_dlc = "Zeus"
			has_law = agnatic_succession
		}
	}
	# Not challenged over infamy
	OR = {
		AND = {
			prestige = 0
			NOR = {
				reverse_has_opinion_modifier = { who = FROM modifier = opinion_tyrant }
				reverse_has_opinion_modifier = { who = FROM modifier = opinion_evil_tyrant }
			}
		}
		has_law = werewolf_law_1
		has_law = werewolf_law_2
		has_law = werewolf_law_3
		has_law = werewolf_law_4
	}
	# Not challenged over weakness
	OR = {
		NOR = {
			trait = weak
			trait = feeble
		}
		has_law = werewolf_law_4
	}
	# Not challenged over congential deformity
	OR = {
		NOR = {
			trait = clubfooted
			trait = harelip
			trait = hunchback
			trait = dwarf
			trait = inbred
		}
		has_law = werewolf_law_3
		has_law = werewolf_law_4
	}
	# Not challenged over short reign
	OR = {
		ruled_years = 5
		AND = {
			ruled_years = 3
			has_law = werewolf_law_4
		}
		AND = {
			ruled_years = 4
			OR = {
				has_law = werewolf_law_4
				has_law = werewolf_law_3
				has_law = werewolf_law_2
			}
		}
	}
	is_incapable = no
}
