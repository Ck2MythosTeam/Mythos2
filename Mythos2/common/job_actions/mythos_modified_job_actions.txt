# Chancellor
action_fabricate_claims = {
	attribute = diplomacy
	
	potential = {
		FROM = { 
			is_nomadic = no 
			NOR = { 
				government = draconic_tribal_government
				government = zombie_horde
			}
		}
	}
	
	trigger = {
		any_province_lord = {
			NOT = { character = FROM }
			FROM = {
				job_chancellor = {
					NOT = { character = PREVPREV } # Can't be sent to improve relations with himself...
				}
			}
		}
		is_within_diplo_range = FROM
	}
	
	events = { 913 914 }
}
